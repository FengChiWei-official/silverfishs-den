`~/.bashrc`

``` bash

############### umu setup start
# 让 umu-run 默认在系统 SSD 里的一个位置管理运行环境，而不是在 NTFS 盘里
export UMU_PREFIX_DIR="$HOME/.local/share/umu_prefixes"

# 如果你需要玩日文游戏，这一行也加上（免去每次输入的麻烦）
alias run-ja="env LC_ALL=ja_JP.UTF-8"
############## umu setup end
```
