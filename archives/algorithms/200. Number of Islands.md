class Solution {

public:

vector<pair<int, int>> dirs = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

  

void dfs(vector<vector<char>>& g, vector<vector<int>>& v, pair<int, int> start) {

int rs = g.size();

int cs = g[0].size();

stack<pair<int, int>> st;

st.push(start);

v[start.first][start.second] = 1; // 1. 入栈即标记

  

while (!st.empty()) {

auto [nx, ny] = st.top(); // 2. 获取当前点

st.pop();

  

for (auto [dx, dy] : dirs) {

int nnx = dx + nx; // 3. 基于当前点偏移

int nny = dy + ny;

  

// 4. 边界 + 内容 + 访问状态 (护城河)

if (nnx >= 0 && nnx < rs && nny >= 0 && nny < cs &&

g[nnx][nny] == '1' && v[nnx][nny] == 0) {

v[nnx][nny] = 1; // 5. 入栈即标记

st.push({nnx, nny});

}

}

}

}

  

int numIslands(vector<vector<char>>& grid) {

if (grid.empty()) return 0;

int gr = grid.size();

int gc = grid[0].size();

  

vector<vector<int>> visited(gr, vector<int>(gc, 0));

int ans = 0;

  

for (int i = 0; i < gr; i++) {

for (int j = 0; j < gc; j++) {

if (grid[i][j] == '1' && visited[i][j] == 0) {

ans++;

dfs(grid, visited, {i, j});

}

}

}

return ans;

}

};
