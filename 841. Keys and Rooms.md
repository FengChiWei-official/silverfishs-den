## BFS

```cpp
class Solution {

public:

int findCircleNum(vector<vector<int>>& isConnected) {

int n = isConnected.size();

vector<int> visited(n, 0);

// 每一次跳转

  

int cnt = 0;

// i 每一次确认有效

  

  

for (int i = 0; i < n; i++) {

  

if (!visited[i]) {

cnt++;

// visited[i] = 1;

queue<int> lvs;

lvs.push(i);

while (!lvs.empty()) {

int cv = lvs.front();

lvs.pop();

visited[cv] = 1;

  

for(int j = 0; j < n; j++) {

int cvj = isConnected[cv][j];

if (cvj == 1 && visited[j] == 0) {

lvs.push(j);

visited[j] = 1;

}

}

  

}

}

  

}

return cnt;

}

};

```
