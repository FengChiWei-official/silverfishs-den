## BFS

```cpp
class Solution {
public:
    int findCircleNum(vector<vector<int>>& isConnected) {
        int n = isConnected.size();
        vector<int> visited(n, 0); 
        int provinces = 0; // cnt -> provinces 语义更明确

        for (int i = 0; i < n; i++) {
            // 如果当前城市未被访问，说明发现了一个新的省份
            if (!visited[i]) {
                provinces++;
                
                queue<int> q; // lvs -> q 
                q.push(i);
                visited[i] = 1; // 【更新时机 1】起点入队即标记

                while (!q.empty()) {
                    int curr = q.front(); // cv -> curr (current)
                    q.pop();

                    for (int next = 0; next < n; next++) {
                        // 如果有连接且邻居未被访问
                        if (isConnected[curr][next] == 1 && !visited[next]) {
                            visited[next] = 1; // 【更新时机 2】邻居入队即标记，防止重复入队
                            q.push(next);
                        }
                    }
                }
            }
        }
        return provinces;
    }
};
```

## DFS


```cpp

class Solution {
public:
    /**
     * @brief 判断是否能访问所有房间 (图的连通性问题)
     * @param rooms 邻接表表示的有向图
     */
    bool canVisitAllRooms(vector<vector<int>>& rooms) {
        int n = rooms.size();
        
        // 1. 初始化状态
        vector<int> vis(n, 0); // 访问标记数组
        int count = 0;         // 已访问节点计数器

        // 2. 启动起点 (0号房间)
        vis[0] = 1; 
        count = 1;
        dfs(0, rooms, vis, count);

        // 3. 判断连通节点数是否等于总节点数
        return count == n;
    }

private:
    /**
     * @brief DFS 递归辅助函数
     * @param curr  当前处理的节点
     * @param g     图的引用 (const 引用避免拷贝)
     * @param vis   标记数组 (引用传递同步状态)
     * @param count 计数器 (引用传递累计结果)
     */
    void dfs(int curr, const vector<vector<int>>& g, vector<int>& vis, int& count) {
        // 使用引用获取邻居列表，效率更高
        const auto& neighbors = g[curr];

        for (const int& nxt : neighbors) {
            // 【核心模板逻辑】：访问前检查标记
            if (!vis[nxt]) {
                vis[nxt] = 1; // 1. 标记即访问
                count++;      // 2. 更新全局结果
                dfs(nxt, g, vis, count); // 3. 递归深入
            }
        }
    }
};
```
